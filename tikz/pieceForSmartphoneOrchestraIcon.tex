\documentclass{standalone}
\usepackage{xcolor}

\usepackage{tikz}
\usetikzlibrary{calc}

% Picture size.
\def\squareSize{10cm}
% Half the diagonal of the inner square
\def\diagonal{2cm}

% Colors
\definecolor{background}{HTML}{222831}
\definecolor{square}{HTML}{F96D00}

\begin{document}
\begin{tikzpicture}
  % Rectangle common center
  \coordinate (K) at (0.5*\squareSize, 0.5*\squareSize);
  % Diagonal distance as point
  \coordinate (D) at (\diagonal, \diagonal);
  \fill [background] (0,0) rectangle (\squareSize, \squareSize);
  \fill [square, rounded corners=2ex] ($(K) - (D)$) rectangle ($(K) + (D)$);
\end{tikzpicture}
\end{document}
