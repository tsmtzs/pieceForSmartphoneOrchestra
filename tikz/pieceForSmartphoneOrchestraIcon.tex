\documentclass{standalone}
\usepackage{fontspec}
\usepackage{xunicode}
\usepackage{xltxtra}
\usepackage{xgreek}
\usepackage{xcolor}

\usepackage{tikz}
\usetikzlibrary{calc}

\setmainfont[Mapping=tex-text]{Linux Biolinum O}

% Picture size.
\def\squareSize{10cm}
% Half the diagonal of the inner square
\def\diagonal{3cm}

% Colors
\definecolor{seashell}{HTML}{FFF5EE}
\definecolor{darkslategray}{HTML}{2F4F4F}

\begin{document}
\begin{tikzpicture}
  % Rectangle common center
  \coordinate (K) at (0.5*\squareSize, 0.5*\squareSize);
  % Diagonal distance as point
  \coordinate (D) at (\diagonal, \diagonal);
  \fill [seashell] (0,0) rectangle (\squareSize, \squareSize);
  \fill [darkslategray, rounded corners=2ex] ($(K) - (D)$) rectangle ($(K) + (D)$);
\end{tikzpicture}
\end{document}
