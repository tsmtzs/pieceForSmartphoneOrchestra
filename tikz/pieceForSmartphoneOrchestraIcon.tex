\documentclass{standalone}
\usepackage{xcolor}

\usepackage{tikz}
\usetikzlibrary{calc}

% Picture size.
\def\squareSize{10cm}
% Half the diagonal of the inner square
\def\diagonal{2cm}

% Colors
\definecolor{background}{HTML}{222831}
\definecolor{color1}{HTML}{F96D00}

\begin{document}
\begin{tikzpicture}
  % Rectangle common center
  \coordinate (K) at (0.5*\squareSize, 0.5*\squareSize);
  % Diagonal distance as point
  \coordinate (D) at (\diagonal, \diagonal);
  \fill [background] (0,0) rectangle (\squareSize, \squareSize);
  \fill [color1, rounded corners=2ex] ($(K) - (D)$) rectangle ($(K) + (D)$);
  % \fill [background, rounded corners=2ex,rotate=45] ($(K) - (D)$) rectangle ($(K) + (D)$);
  \fill [background,rounded corners=2ex,transform canvas={rotate around={45:(K)}}] ($(K) - 1*(D)$) rectangle ($(K) + 1*(D)$);
  \fill [background,rounded corners=2ex,transform canvas={rotate around={0:(K)}}] ($(K) - 0.95*(D)$) rectangle ($(K) + 0.95*(D)$);
  \fill [color1,rounded corners=2ex] ($(K) - 0.8*(D)$) rectangle ($(K) + 0.8*(D)$);
  \draw [background,rounded corners=2ex,line width=0.05cm] ($(K) - 0.72*(D)$) rectangle ($(K) + 0.72*(D)$);
\end{tikzpicture}
\end{document}
